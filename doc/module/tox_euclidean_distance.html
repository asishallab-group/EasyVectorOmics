public pure subroutine euclidean_distance(vec1, vec2, d, result)

Compute the Euclidean distance between two vectors. Calculates the L2 norm: result = sqrt(sum((vec1_i - vec2_i)^2))

Arguments

Output scalar distance

public pure subroutine distance_to_centroid(n_genes, n_families, genes, centroids, gene_to_fam, distances, d)

Compute distance from each gene to its corresponding family centroid. For each gene, extracts its expression vector and the centroid of its assigned family, then computes the Euclidean distance between them.

Arguments

Type IntentOptional Attributes Name
integer(kind=int32), intent(in) :: n_genes

real(kind=real64), intent(in) :: centroids(d,n_families)
